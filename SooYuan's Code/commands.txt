extends Control

# List of commands you want to display
var commands = [
	"/h - Display commands",
	"/roll - Roll a dice",
	"/spawn x - Generate a X",
	"/spawn X - Generate a X",
	"/spawn o - Generate a O",
	"/spawn O - Generate a O"
]

func _ready():
	# Get the VBoxContainer node
	var vbox = $Panel/VBoxContainer
	
	# Add each command as a Label
	for command_text in commands:
		var label = Label.new()
		label.text = command_text
		label.autowrap_mode = TextServer.AUTOWRAP_WORD
		vbox.add_child(label)
